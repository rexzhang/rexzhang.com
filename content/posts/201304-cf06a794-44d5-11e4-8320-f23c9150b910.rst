[code:python]
#!/usr/bin/env python
#coding=utf-8

uuu = u'中文'
print type(uuu), uuu

sss = uuu.encode('gbk')
print type(sss), sss

nnn = unicode(sss, 'gbk')
print type(nnn), nnn
[/code]

输出为
[code:text]
<type 'unicode'> 中文
<type 'str'> 中文
<type 'unicode'> 中文
[/code]
